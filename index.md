    <script type="text/babel">
        const WrestlingBracket = () => {
            const [selectedDivisionWeight, setSelectedDivisionWeight] = React.useState('');
            const [divisions, setDivisions] = React.useState([]);
            const [wrestlers, setWrestlers] = React.useState([]);
            
            const sortDivisions = (divisions) => {
                const orderMap = {
                    'Girls 6U': 1,
                    'Girls 8U': 2,
                    'Girls 10U': 3,
                    'Girls 12U': 4,
                    'Girls 15U': 5,
                    'Open 6U': 6,
                    'Open 8U': 7,
                    'Open 10U': 8,
                    'Open 12U': 9,
                    'Open 15U': 10
                };

                return divisions.sort((a, b) => {
                    const [divA, weightA] = a.split(' - ');
                    const [divB, weightB] = b.split(' - ');
                    
                    if (orderMap[divA] !== orderMap[divB]) {
                        return orderMap[divA] - orderMap[divB];
                    }
                    
                    if (weightA === 'HWT') return 1;
                    if (weightB === 'HWT') return -1;
                    
                    const numA = parseInt(weightA);
                    const numB = parseInt(weightB);
                    
                    return numA - numB;
                });
            };

            React.useEffect(() => {
                const xmlText = document.getElementById('wrestlerData').textContent;
                const parser = new DOMParser();
                const xmlDoc = parser.parseFromString(xmlText, 'text/xml');
                const wrestlerNodes = xmlDoc.getElementsByTagName('wrestler');
                
                const wrestlerData = Array.from(wrestlerNodes).map(node => ({
                    name: node.getElementsByTagName('name')[0]?.textContent || '',
                    divisionWeight: node.getElementsByTagName('divisionWeight')[0]?.textContent || '',
                    seed: parseInt(node.getElementsByTagName('seed')[0]?.textContent) || 0
                }));

                const uniqueDivisions = [...new Set(wrestlerData.map(w => w.divisionWeight))];
                const sortedDivisions = sortDivisions(uniqueDivisions);
                
                setDivisions(sortedDivisions);
                setSelectedDivisionWeight(sortedDivisions[0] || '');
                setWrestlers(wrestlerData);
            }, []);

            const filteredWrestlers = wrestlers
                .filter(w => w.divisionWeight === selectedDivisionWeight)
                .sort((a, b) => a.seed - b.seed);

            const createMatchesWithByes = (wrestlers) => {
                const seedOrder = [
                    1, null, 17, 16, 9, 24, null, 8,
                    5, null, 21, 12, 13, 20, null, 4,
                    3, null, 19, 14, 11, 22, null, 6,
                    7, null, 23, 10, 15, 18, null, 2
                ];

                return seedOrder.map(seed => {
                    if (seed === null) {
                        return { id: null, name: "Bye" };
                    }
                    const wrestler = wrestlers.find(w => w.seed === seed);
                    return {
                        id: seed,
                        name: wrestler ? wrestler.name : `Not Seeded`
                    };
                });
            };

            const matches = createMatchesWithByes(filteredWrestlers);

            const MatchRow = ({ match, showConnector, connectorHeight }) => (
                <div className="flex items-center h-6 relative">
                    <div className="w-8 text-right pr-2">
                        {match.id && match.id}
                    </div>
                    <div className="w-64 border border-gray-300 px-1 text-sm overflow-hidden whitespace-nowrap">
                        {match.name}
                    </div>
                    <div className="h-px w-4 border-t border-gray-300"></div>
                    {showConnector && (
                        <div 
                            className="absolute border-r border-gray-300"
                            style={{
                                right: 0,
                                top: '12px',
                                height: `${connectorHeight}px`
                            }}
                        />
                    )}
                </div>
            );

            return (
                <div className="p-4">
                    <div className="mb-4">
                        <label className="mr-2 font-semibold">Division/Weight:</label>
                        <select 
                            value={selectedDivisionWeight}
                            onChange={(e) => setSelectedDivisionWeight(e.target.value)}
                            className="border border-gray-300 rounded px-2 py-1 w-48"
                        >
                            {divisions.map(division => (
                                <option key={division} value={division}>
                                    {division}
                                </option>
                            ))}
                        </select>
                    </div>
                    
                    <div className="font-mono">
                        {matches.map((match, index) => {
                            let showConnector = false;
                            let connectorHeight = 0;
                            
                            if (index % 2 === 0 && index < matches.length - 1) {
                                showConnector = true;
                                if (index % 4 === 0) {
                                    connectorHeight = 24;
                                } else if (index % 8 === 0) {
                                    connectorHeight = 48;
                                } else if (index % 16 === 0) {
                                    connectorHeight = 96;
                                }
                            }

                            return (
                                <div key={index} className="relative">
                                    <MatchRow 
                                        match={match}
                                        showConnector={showConnector}
                                        connectorHeight={connectorHeight}
                                    />
                                </div>
                            );
                        })}
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<WrestlingBracket />);
    </script>
